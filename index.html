function processarArquivos(files){ Array.from(files).forEach(file=>{ let reader=new FileReader(); reader.onload=e=>{ let base64 = e.target.result; // Criar imagem para leitura EXIF correta let img = new Image(); img.onload = function(){ EXIF.getData(img, function(){ let lat=convertExifToDecimal(EXIF.getTag(this,"GPSLatitude"),EXIF.getTag(this,"GPSLatitudeRef")); let lon=convertExifToDecimal(EXIF.getTag(this,"GPSLongitude"),EXIF.getTag(this,"GPSLongitudeRef")); // 📂 SE VEIO DA GALERIA if(fileInput.files.length > 0){ if(lat && lon){ renderizarFotoHtml(base64,lat,lon,"Metadados EXIF"); }else{ renderizarFotoHtml(base64,null,null,"Localização não encontrada nos metadados da imagem"); } } // 📷 SE VEIO DA CÂMERA else{ if(lat && lon){ renderizarFotoHtml(base64,lat,lon,"EXIF da câmera"); }else{ navigator.geolocation.getCurrentPosition( pos=>{ renderizarFotoHtml(base64,pos.coords.latitude,pos.coords.longitude,"GPS do dispositivo"); }, err=>{ renderizarFotoHtml(base64,null,null,"Sem localização disponível"); } ); } } }); }; img.src = base64; }; reader.readAsDataURL(file); }); // limpa inputs fileInput.value = ''; cameraInput.value = ''; }
